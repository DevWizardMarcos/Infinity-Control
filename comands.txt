"systemctl mask sleep.target suspend.target hibernate.target hybrid-sleep.target"

# Manter o computador acordado por X horas
"systemd-inhibit --what=sleep --who='SSH Session' --why='Prevent sleep' sleep 3600"

"echo 'enabled' > /sys/class/rtc/rtc0/wakealarm"
